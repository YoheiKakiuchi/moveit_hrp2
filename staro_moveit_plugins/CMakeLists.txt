cmake_minimum_required(VERSION 2.8.3)
project(staro_moveit_plugins)

if(NOT CMAKE_CONFIGURATION_TYPES AND NOT CMAKE_BUILD_TYPE)
  set(CMAKE_BUILD_TYPE Release)
endif()

find_package(Eigen REQUIRED)
find_package(Boost REQUIRED thread)

find_package(catkin COMPONENTS
  moveit_core
  moveit_msgs
  pluginlib
  roscpp
  REQUIRED
)

catkin_package(
  LIBRARIES
  INCLUDE_DIRS
    staro_moveit_constraint_sampler/include/
  DEPENDS
    moveit_core
)

include_directories(SYSTEM ${EIGEN_INCLUDE_DIRS} ${Boost_INCLUDE_DIR})
include_directories(
  staro_moveit_constraint_sampler/include/
  ${catkin_INCLUDE_DIRS}
)

link_directories(${Boost_LIBRARY_DIRS})
link_directories(${catkin_LIBRARY_DIRS})

add_subdirectory(staro_moveit_constraint_sampler)

install(
  FILES
    staro_moveit_constraint_sampler_plugin_description.xml
  DESTINATION
    ${CATKIN_PACKAGE_SHARE_DESTINATION}
)
